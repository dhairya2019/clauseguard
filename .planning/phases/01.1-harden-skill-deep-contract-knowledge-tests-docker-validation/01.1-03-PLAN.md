**tests/run-tests.sh** -- Main test runner:
```bash
#!/usr/bin/env bash
set -euo pipefail

# Configuration
CONTRACT_DIR="./tests/contracts"
EXPECTED_DIR="./tests/expected"
OUTPUT_DIR="./tests/output"
PASS=0; FAIL=0; WARN=0; SKIP=0
ERRORS=()

mkdir -p "$OUTPUT_DIR"

# Check prerequisites
if ! command -v claude &> /dev/null; then
  echo "ERROR: claude CLI not found. Install: npm install -g @anthropic-ai/claude-code"
  exit 1
fi

if [ -z "${ANTHROPIC_API_KEY:-}" ]; then
  echo "ERROR: ANTHROPIC_API_KEY not set"
  exit 1
fi

echo "=== ClauseGuard Test Suite ==="
echo "Date: $(date)"
echo "Contracts: $(ls "$CONTRACT_DIR"/*.txt 2>/dev/null | wc -l | tr -d ' ')"
echo ""

for contract in "$CONTRACT_DIR"/*.txt; do
  name=$(basename "$contract" .txt)
  expected="$EXPECTED_DIR/${name}.expected"
  output="$OUTPUT_DIR/${name}.output"

  if [ ! -f "$expected" ]; then
    echo "SKIP: $name (no .expected file)"
    ((SKIP++))
    continue
  fi

  echo "Testing: $name"

  # Run ClauseGuard skill in headless mode
  # --max-turns 1 limits to single response (cost control)
  # Redirect stderr to suppress progress output
  if ! claude -p "/clauseguard $contract" \
    --output-format text \
    --max-turns 1 \
    > "$output" 2>/dev/null; then
    echo "  ERROR: claude command failed for $name"
    ERRORS+=("$name: claude invocation failed")
    ((FAIL++))
    continue
  fi

  # Validate output against expected markers
  source ./tests/validate.sh "$output" "$expected" name PASS FAIL WARN ERRORS
done

echo ""
echo "=== Results ==="
echo "PASS: $PASS"
echo "FAIL: $FAIL"
echo "WARN: $WARN"
echo "SKIP: $SKIP"

if [ ${#ERRORS[@]} -gt 0 ]; then
  echo ""
  echo "=== Failures ==="
  for err in "${ERRORS[@]}"; do
    echo "  - $err"
  done
fi

echo ""
if [ $FAIL -gt 0 ]; then
  echo "RESULT: FAILED ($FAIL failures)"
  exit 1
else
  echo "RESULT: PASSED"
  exit 0
fi
```

NOTE: The above is a structural guide. The actual implementation should handle edge cases like empty output files and use proper bash variable passing for the validate function. The validate.sh should be sourced or called as a function.

**tests/validate.sh** -- Marker validation logic:
A bash script/function that reads an .expected file and validates each MUST/SHOULD/MUSTNOT pattern against the output file using case-insensitive grep.

```bash
#!/usr/bin/env bash
# Usage: validate.sh <output_file> <expected_file>
# Returns: exit code 0 if all MUST pass, 1 if any MUST fails

OUTPUT="$1"
EXPECTED="$2"
LOCAL_PASS=0; LOCAL_FAIL=0; LOCAL_WARN=0

while IFS= read -r line; do
  # Skip comments and empty lines
  [[ "$line" =~ ^#.*$ || -z "$line" ]] && continue

  type="${line%%:*}"
  pattern="${line#*:}"

  case "$type" in
    MUST)
      if ! grep -qi "$pattern" "$OUTPUT"; then
        echo "  FAIL: MUST pattern not found: $pattern"
        ((LOCAL_FAIL++))
      else
        ((LOCAL_PASS++))
      fi
      ;;
    SHOULD)
      if ! grep -qi "$pattern" "$OUTPUT"; then
        echo "  WARN: SHOULD pattern not found: $pattern"
        ((LOCAL_WARN++))
      else
        ((LOCAL_PASS++))
      fi
      ;;
    MUSTNOT)
      if grep -qi "$pattern" "$OUTPUT"; then
        echo "  FAIL: MUSTNOT pattern found: $pattern"
        ((LOCAL_FAIL++))
      else
        ((LOCAL_PASS++))
      fi
      ;;
  esac
done < "$EXPECTED"

echo "  Result: $LOCAL_PASS pass, $LOCAL_FAIL fail, $LOCAL_WARN warn"
exit $LOCAL_FAIL
```

IMPORTANT: The run-tests.sh should call validate.sh per contract (not source it).
